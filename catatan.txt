SELECT pembayaran.INVOICE, termin_pks.NO_PKS, pembayaran.TGL_INVOICE, termin_pks.TERMIN, termin_pks.NOMINAL, pks.NOMINAL_PKS
FROM termin_pks, pembayaran, pks
WHERE termin_pks.KODE_TERMIN = pembayaran.KODE_TERMIN AND termin_pks.NO_PKS = pks.NO_PKS
